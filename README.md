        System.out.println("Сколько элементов последовательности вывести: ");

        for (int i = 1; i <= n; i++) {

            if (i % 2 == 0) {
                a += 2;
            }

            else {
                a *= 3;
            }

            System.out.print(a);

            if (i < n) {
                System.out.print(", ");
            }
        }
    }
}
